235 public static void AddCodeBlockToOB(XmlLogicalBlock codeBlock, ResourceManagement resourceManagement, string languagesettings, PlcDevice plcDevice)
236 {
237 var existingOB = OpennessFuncs.TryGetOBFromPlcDevice(plcDevice, codeBlock.Name);
238 //Create new OB in SCL with calling code
239 if (existingOB == null)
240 {
241 if (codeBlock.BlockAttributes.ProgrammingLanguage == ProgrammingLanguage.SCL)
242 {
243 codeBlock.GenerateSclBlock(plcDevice);
244 }
245 else //insert networks to KOP/FUP
246 {
247 codeBlock.GenerateXmlBlock(plcDevice, null, languagesettings);
248 }
249 }
250 else
251 //Insert Code to existing OB
252 {
253 var blockAsDoc = OpennessFuncs.ExportBlockAsXml(existingOB.Name, plcDevice, OpennessFuncs.CachingOptions.DISABLED);
254 Parser parser = new Parser();
255 var xmlOB = parser.ParseXml(blockAsDoc);
256 xmlOB.MergeBlock(codeBlock);
257 //add SCL-Code
258 if (xmlOB.BlockAttributes.ProgrammingLanguage == ProgrammingLanguage.SCL)
259 {
260 xmlOB.GenerateSclBlock(plcDevice);
261 }
262 else //insert networks to KOP/FUP
263 {
264 xmlOB.GenerateXmlBlock(plcDevice, null, languagesettings);
265 }
266 }
